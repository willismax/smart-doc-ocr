## 需求分析與設計約束 {#需求分析}

### 硬需求（不可妥協）

| 需求 | 說明 | 技術影響 |
|------|------|---------|
| **完全離線** | 不呼叫任何外部 API | 所有模型須預先下載；禁用雲端 OCR |
| **個資不外洩** | 文件含身分證、姓名、帳號等 | 無網路呼叫、本地加密、稽核日誌 |
| **CPU 可執行** | 不強制需要 GPU | OCR 引擎需支援 CPU 模式 |
| **文書人員操作** | 非技術人員 | UI 需極簡，不露出 CLI |
| **Windows 11** | 企業部署環境 | 避免 Linux-only 依賴 |

### 軟需求（可彈性調整）

- 支援格式：PDF（數位/掃描）、Word、Excel、PPT、圖片（JPG/PNG/TIFF/BMP）
- 批量：一次可丟入 100+ 份文件
- 比對模式：文字差異、欄位比對、語意相似度
- 輸出：Markdown 報告、Excel 差異表

## 技術選型與理由 {#技術選型}

### OCR 引擎：PaddleOCR v4（首選）

```
選擇理由：
✅ 繁體中文、簡體中文支援極佳（比 Tesseract 高 20~30% 準確率）
✅ 完全離線，模型可打包
✅ 支援 CPU / GPU 自動切換
✅ 內建版面分析（PP-Structure）：表格、標題、段落自動分類
✅ 角度校正（旋轉文件自動扶正）
✅ 支援 Windows

備選：
- Tesseract 5.x：更輕量，中文較弱
- EasyOCR：易用但速度慢
- docTR：架構優雅，對 GPU 加速較依賴
```

### 文件讀取層

| 格式 | 工具 | 說明 |
|------|------|------|
| PDF（數位，有文字層）| PyMuPDF (fitz) | 直接萃取，速度快 |
| PDF（掃描，無文字層）| pdf2image → PaddleOCR | 轉圖片後 OCR |
| PDF（混合）| 自動偵測文字覆蓋率決定路徑 | |
| Word (.docx) | MarkItDown + python-docx | 保留格式結構 |
| Excel (.xlsx) | MarkItDown + openpyxl | 表格轉 Markdown |
| PPT (.pptx) | MarkItDown | 投影片文字萃取 |
| 圖片 (JPG/PNG/TIFF) | Pillow + PaddleOCR | 前處理後 OCR |
| Email (.eml/.msg) | extract-msg + mailparser | 含附件遞迴處理 |

### Layer 1：文件路由器

```python
# document_router.py
import magic  # python-magic-bin（Windows 版）
from pathlib import Path
from enum import Enum

class DocType(Enum):
    PDF_DIGITAL  = "pdf_digital"   # PDF 有文字層
    PDF_SCANNED  = "pdf_scanned"   # PDF 掃描，無文字
    PDF_MIXED    = "pdf_mixed"     # 部分頁面有文字
    WORD         = "word"
    EXCEL        = "excel"
    PPT          = "ppt"
    IMAGE        = "image"
    EMAIL        = "email"
    UNKNOWN      = "unknown"

class DocumentRouter:
    # 文字覆蓋率低於此閾值 → 視為掃描 PDF
    TEXT_COVERAGE_THRESHOLD = 0.1
    
    def detect(self, file_path: Path) -> DocType:
        """用 magic bytes 偵測真實格式（不信任副檔名）"""
        mime = magic.from_file(str(file_path), mime=True)
        
        if mime == "application/pdf":
            return self._classify_pdf(file_path)
        elif mime in ("application/vnd.openxmlformats-officedocument"
                      ".wordprocessingml.document", "application/msword"):
            return DocType.WORD
        elif mime in ("application/vnd.openxmlformats-officedocument"
                      ".spreadsheetml.sheet", "application/vnd.ms-excel"):
            return DocType.EXCEL
        elif mime in ("application/vnd.openxmlformats-officedocument"
                      ".presentationml.presentation",):
            return DocType.PPT
        elif mime.startswith("image/"):
            return DocType.IMAGE
        elif mime in ("message/rfc822", "application/vnd.ms-outlook"):
            return DocType.EMAIL
        return DocType.UNKNOWN

    def _classify_pdf(self, file_path: Path) -> DocType:
        """分析 PDF 文字覆蓋率，決定是否需要 OCR"""
        import fitz
        doc = fitz.open(str(file_path))
        total_pages = len(doc)
        text_pages = 0
        
        for page in doc:
            text = page.get_text().strip()
            if len(text) > 50:  # 有實質文字
                text_pages += 1
        
        doc.close()
        coverage = text_pages / max(total_pages, 1)
        
        if coverage > 0.8:
            return DocType.PDF_DIGITAL
        elif coverage < self.TEXT_COVERAGE_THRESHOLD:
            return DocType.PDF_SCANNED
        else:
            return DocType.PDF_MIXED
```

### Layer 2：前處理（影像品質增強）

```python
# preprocessor.py
import cv2
import numpy as np
from PIL import Image

class ImagePreprocessor:
    """OCR 前影像增強，顯著提升低品質掃描件辨識率"""
    
    def process(self, image: np.ndarray) -> np.ndarray:
        """完整前處理流程"""
        img = self._to_gray(image)
        img = self._deskew(img)          # 歪斜校正
        img = self._denoise(img)         # 去雜訊
        img = self._enhance_contrast(img)# 對比度增強
        img = self._binarize(img)        # 二值化（黑白）
        return img

    def _to_gray(self, img):
        if len(img.shape) == 3:
            return cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        return img
    
    def _deskew(self, img):
        """偵測並校正文件傾斜角度"""
        coords = np.column_stack(np.where(img < 128))
        if len(coords) == 0:
            return img
        angle = cv2.minAreaRect(coords)[-1]
        if angle < -45:
            angle = 90 + angle
        if abs(angle) < 0.5:  # 傾斜不到 0.5 度不處理
            return img
        h, w = img.shape
        M = cv2.getRotationMatrix2D((w // 2, h // 2), angle, 1.0)
        return cv2.warpAffine(img, M, (w, h),
                              flags=cv2.INTER_CUBIC,
                              borderMode=cv2.BORDER_REPLICATE)
    
    def _denoise(self, img):
        """去除掃描雜訊"""
        return cv2.fastNlMeansDenoising(img, h=10)
    
    def _enhance_contrast(self, img):
        """CLAHE 自適應對比度增強"""
        clahe = cv2.createCLAHE(clipLimit=2.0, tileGridSize=(8, 8))
        return clahe.apply(img)
    
    def _binarize(self, img):
        """Otsu 自動二值化"""
        _, binary = cv2.threshold(img, 0, 255,
                                   cv2.THRESH_BINARY + cv2.THRESH_OTSU)
        return binary
    
    def scale_for_ocr(self, img, target_dpi=300):
        """確保解析度足夠（至少 300 DPI）"""
        h, w = img.shape[:2]
        # 假設原始 72 DPI，放大到 300 DPI
        scale = target_dpi / 72
        new_w = int(w * scale)
        new_h = int(h * scale)
        if scale > 1:
            return cv2.resize(img, (new_w, new_h), interpolation=cv2.INTER_CUBIC)
        return img
```

### Layer 3：辨識層（OCR + 結構化萃取）

```python
# recognizer.py
from paddleocr import PaddleOCR, PPStructure
import fitz
from markitdown import MarkItDown
from pathlib import Path
import numpy as np

class DocumentRecognizer:
    def __init__(self, use_gpu: bool = False, lang: str = "ch"):
        # lang="ch" 支援繁體+簡體中文
        self.ocr = PaddleOCR(
            use_angle_cls=True,   # 角度分類
            lang=lang,
            use_gpu=use_gpu,
            show_log=False,
            # 離線模式：指定本地模型路徑
            det_model_dir="./models/det",
            rec_model_dir="./models/rec",
            cls_model_dir="./models/cls",
        )
        # PP-Structure：版面分析（表格、標題、段落）
        self.structure = PPStructure(
            table=True,
            ocr=True,
            show_log=False,
            image_orientation=True,
        )
        self.markitdown = MarkItDown()
    
    def extract_pdf_digital(self, path: Path) -> dict:
        """數位 PDF：直接萃取文字，保留結構"""
        doc = fitz.open(str(path))
        pages = []
        
        for page_num, page in enumerate(doc):
            # 萃取文字與位置
            text_dict = page.get_text("dict")
            text = page.get_text()
            
            # 萃取表格（如有）
            tables = page.find_tables()
            table_data = []
            for table in tables:
                table_data.append(table.extract())
            
            pages.append({
                "page": page_num + 1,
                "text": text.strip(),
                "tables": table_data,
                "bbox": page.rect,
            })
        
        doc.close()
        return {"type": "pdf_digital", "pages": pages, "path": str(path)}
    
    def extract_pdf_scanned(self, path: Path) -> dict:
        """掃描 PDF：轉圖片後 OCR"""
        from pdf2image import convert_from_path
        preprocessor = ImagePreprocessor()
        pages = []
        
        # 轉換為高解析度圖片
        images = convert_from_path(str(path), dpi=300, fmt="png")
        
        for page_num, pil_image in enumerate(images):
            import numpy as np
            img_array = np.array(pil_image)
            
            # 前處理
            processed = preprocessor.process(img_array)
            
            # OCR（PP-Structure 同時做版面分析）
            result = self.structure(processed)
            
            text_blocks = []
            tables = []
            
            for region in result:
                if region["type"] == "text":
                    text_blocks.append(region.get("res", {}).get("text", ""))
                elif region["type"] == "table":
                    tables.append(region.get("res", {}).get("html", ""))
            
            pages.append({
                "page": page_num + 1,
                "text": "\n".join(text_blocks),
                "tables": tables,
            })
        
        return {"type": "pdf_scanned", "pages": pages, "path": str(path)}
    
    def extract_office(self, path: Path) -> dict:
        """Office 格式：MarkItDown 轉換"""
        result = self.markitdown.convert(str(path))
        return {
            "type": "office",
            "text": result.text_content,
            "path": str(path)
        }
    
    def extract_image(self, path: Path) -> dict:
        """純圖片 OCR"""
        import cv2
        preprocessor = ImagePreprocessor()
        img = cv2.imread(str(path))
        processed = preprocessor.process(img)
        
        result = self.ocr.ocr(processed, cls=True)
        
        texts = []
        if result and result[0]:
            for line in result[0]:
                texts.append(line[1][0])  # 文字內容
        
        return {
            "type": "image",
            "text": "\n".join(texts),
            "path": str(path)
        }
```

### Layer 4：結構化層（清洗與格式化）

```python
# structurer.py
import re

class TextStructurer:
    """把 OCR 原始輸出清洗成乾淨的 Markdown"""
    
    def structure(self, raw: dict) -> str:
        doc_type = raw.get("type")
        
        if doc_type in ("pdf_digital", "pdf_scanned"):
            return self._structure_pages(raw["pages"])
        elif doc_type == "office":
            return self._clean_text(raw["text"])
        elif doc_type == "image":
            return self._clean_text(raw["text"])
        return raw.get("text", "")
    
    def _structure_pages(self, pages: list) -> str:
        parts = []
        for p in pages:
            parts.append(f"\n## 第 {p['page']} 頁\n")
            
            if p.get("text"):
                parts.append(self._clean_text(p["text"]))
            
            for i, table in enumerate(p.get("tables", [])):
                parts.append(f"\n### 表格 {i+1}\n")
                parts.append(self._table_to_markdown(table))
        
        return "\n".join(parts)
    
    def _clean_text(self, text: str) -> str:
        """OCR 常見雜訊清洗"""
        # 移除過多空白行
        text = re.sub(r'\n{3,}', '\n\n', text)
        # 修正常見 OCR 錯誤（可依文件類型擴充）
        ocr_corrections = {
            '０': '0', '１': '1', '２': '2', '３': '3',
            '４': '4', '５': '5', '６': '6', '７': '7',
            '８': '8', '９': '9',
            '—': '-', '‒': '-',
        }
        for wrong, right in ocr_corrections.items():
            text = text.replace(wrong, right)
        return text.strip()
    
    def _table_to_markdown(self, table_data) -> str:
        """二維陣列轉 Markdown 表格"""
        if not table_data or not table_data[0]:
            return ""
        
        header = table_data[0]
        rows = table_data[1:]
        
        md = "| " + " | ".join(str(h) for h in header) + " |\n"
        md += "| " + " | ".join("---" for _ in header) + " |\n"
        for row in rows:
            md += "| " + " | ".join(str(c) for c in row) + " |\n"
        
        return md
```

### Layer 5：個資保護層

```python
# pii_protector.py
from presidio_analyzer import AnalyzerEngine, Pattern, PatternRecognizer
from presidio_anonymizer import AnonymizerEngine
from presidio_anonymizer.entities import OperatorConfig
import json
from datetime import datetime
from pathlib import Path

class PIIProtector:
    def __init__(self, audit_log_path: Path = Path("./logs/audit.jsonl")):
        self.analyzer = AnalyzerEngine()
        self.anonymizer = AnonymizerEngine()
        self.audit_log_path = audit_log_path
        self.audit_log_path.parent.mkdir(exist_ok=True)
        
        # 新增台灣身分證識別器
        self._add_tw_recognizers()
    
    def _add_tw_recognizers(self):
        """新增台灣特有的 PII 模式"""
        # 身分證字號：1 英文 + 9 數字
        tw_id = PatternRecognizer(
            supported_entity="TW_ID_NUMBER",
            patterns=[Pattern(
                name="tw_id",
                regex=r'[A-Z][12]\d{8}',
                score=0.85
            )]
        )
        # 手機號碼
        tw_phone = PatternRecognizer(
            supported_entity="TW_PHONE",
            patterns=[Pattern(
                name="tw_mobile",
                regex=r'09\d{8}',
                score=0.8
            )]
        )
        self.analyzer.registry.add_recognizer(tw_id)
        self.analyzer.registry.add_recognizer(tw_phone)
    
    def analyze_and_mask(self, text: str, file_name: str,
                          operator: str = "replace") -> dict:
        """
        分析 PII 並遮蔽
        operator 選項：
          "replace" → 替換為 <ENTITY_TYPE>（預設）
          "redact"  → 完全刪除
          "hash"    → SHA-256 雜湊（可逆查詢）
          "keep"    → 保留（記錄但不遮蔽）
        """
        results = self.analyzer.analyze(
            text=text,
            language="zh",
            entities=[
                "PERSON", "EMAIL_ADDRESS", "PHONE_NUMBER",
                "CREDIT_CARD", "IBAN_CODE", "IP_ADDRESS",
                "TW_ID_NUMBER", "TW_PHONE"
            ]
        )
        
        operators = {entity.entity_type: OperatorConfig(operator)
                     for entity in results}
        
        anonymized = self.anonymizer.anonymize(
            text=text,
            analyzer_results=results,
            operators=operators,
        )
        
        # 寫入稽核日誌
        self._write_audit(file_name, results)
        
        return {
            "original_text": text,
            "masked_text": anonymized.text,
            "pii_count": len(results),
            "pii_types": list(set(r.entity_type for r in results)),
        }
    
    def _write_audit(self, file_name: str, results: list):
        """不可竄改的稽核日誌"""
        log_entry = {
            "timestamp": datetime.now().isoformat(),
            "file": file_name,
            "pii_detected": [
                {"type": r.entity_type, "score": round(r.score, 2)}
                for r in results
            ]
        }
        with open(self.audit_log_path, "a", encoding="utf-8") as f:
            f.write(json.dumps(log_entry, ensure_ascii=False) + "\n")
```

### Layer 6：比對引擎

```python
# comparator.py
import difflib
from sentence_transformers import SentenceTransformer
from sklearn.metrics.pairwise import cosine_similarity
import numpy as np

class DocumentComparator:
    def __init__(self, model_path: str = "./models/multilingual-MiniLM"):
        # 離線語意模型（預先下載）
        self.semantic_model = SentenceTransformer(model_path)
    
    # ── 模式一：逐行文字差異 ──────────────────────────────────
    def diff_text(self, text_a: str, text_b: str) -> dict:
        """產生 unified diff 格式的差異報告"""
        lines_a = text_a.splitlines(keepends=True)
        lines_b = text_b.splitlines(keepends=True)
        
        diff = list(difflib.unified_diff(
            lines_a, lines_b,
            fromfile="文件 A", tofile="文件 B", lineterm=""
        ))
        
        matcher = difflib.SequenceMatcher(None, text_a, text_b)
        similarity = round(matcher.ratio(), 4)
        
        return {
            "similarity_ratio": similarity,
            "similarity_pct": f"{similarity * 100:.1f}%",
            "diff_lines": diff,
            "added": sum(1 for l in diff if l.startswith("+")),
            "removed": sum(1 for l in diff if l.startswith("-")),
        }
    
    # ── 模式二：語意相似度 ────────────────────────────────────
    def semantic_similarity(self, text_a: str, text_b: str) -> float:
        """
        不看字面，看意思是否相近
        回傳 0.0（完全不同）~ 1.0（完全相同）
        """
        # 長文件切段取平均
        chunks_a = self._chunk(text_a)
        chunks_b = self._chunk(text_b)
        
        emb_a = self.semantic_model.encode(chunks_a)
        emb_b = self.semantic_model.encode(chunks_b)
        
        # 取文件整體向量（平均池化）
        vec_a = np.mean(emb_a, axis=0, keepdims=True)
        vec_b = np.mean(emb_b, axis=0, keepdims=True)
        
        score = cosine_similarity(vec_a, vec_b)[0][0]
        return round(float(score), 4)
    
    # ── 模式三：欄位比對（適合表單/合約） ────────────────────
    def field_compare(self, doc_a: dict, doc_b: dict,
                       field_map: dict) -> list:
        """
        field_map 範例：
        {
            "姓名": {"type": "exact"},        # 完全相符
            "金額": {"type": "numeric"},       # 數字比對
            "地址": {"type": "fuzzy", "threshold": 0.8}  # 模糊比對
        }
        """
        results = []
        for field, config in field_map.items():
            val_a = doc_a.get(field, "")
            val_b = doc_b.get(field, "")
            
            if config["type"] == "exact":
                match = val_a == val_b
                score = 1.0 if match else 0.0
            elif config["type"] == "numeric":
                num_a = self._extract_number(val_a)
                num_b = self._extract_number(val_b)
                match = num_a == num_b
                score = 1.0 if match else 0.0
            elif config["type"] == "fuzzy":
                score = difflib.SequenceMatcher(None, val_a, val_b).ratio()
                match = score >= config.get("threshold", 0.8)
            
            results.append({
                "field": field,
                "value_a": val_a,
                "value_b": val_b,
                "match": match,
                "score": round(score, 3),
            })
        
        return results
    
    def _chunk(self, text: str, size: int = 512) -> list:
        """把長文件切成小段，避免超出模型長度限制"""
        words = text.split()
        return [" ".join(words[i:i+size]) for i in range(0, len(words), size)]
    
    def _extract_number(self, text: str) -> float:
        """從字串中抽取數字"""
        import re
        nums = re.findall(r'[\d,.]+', str(text).replace(",", ""))
        return float(nums[0]) if nums else None
```

### 錯誤隔離原則

**每份文件獨立處理，一份失敗不影響整批。**

```python
# batch_processor.py
from dataclasses import dataclass, field
from enum import Enum
import traceback
import time

class Status(Enum):
    PENDING   = "pending"
    RUNNING   = "running"
    SUCCESS   = "success"
    FAILED    = "failed"
    SKIPPED   = "skipped"

@dataclass
class ProcessingResult:
    file_path: str
    status: Status
    output: dict = field(default_factory=dict)
    error: str = ""
    duration_sec: float = 0.0
    retries: int = 0

def process_with_retry(file_path: Path, pipeline,
                        max_retries: int = 3,
                        backoff: float = 2.0) -> ProcessingResult:
    """帶重試和退避的文件處理"""
    start = time.time()
    
    for attempt in range(max_retries):
        try:
            result = pipeline.process(file_path)
            return ProcessingResult(
                file_path=str(file_path),
                status=Status.SUCCESS,
                output=result,
                duration_sec=round(time.time() - start, 2),
                retries=attempt,
            )
        
        except MemoryError:
            # 記憶體不足：不重試，直接標記跳過
            return ProcessingResult(
                file_path=str(file_path),
                status=Status.SKIPPED,
                error="記憶體不足，請分批處理或增加記憶體",
            )
        
        except Exception as e:
            error_msg = traceback.format_exc()
            if attempt < max_retries - 1:
                time.sleep(backoff ** attempt)  # 指數退避
            else:
                return ProcessingResult(
                    file_path=str(file_path),
                    status=Status.FAILED,
                    error=error_msg,
                    duration_sec=round(time.time() - start, 2),
                    retries=attempt + 1,
                )
```

## 批量處理架構 {#批量處理}

```python
# batch_runner.py
from concurrent.futures import ThreadPoolExecutor, as_completed
from pathlib import Path
from rich.progress import Progress, SpinnerColumn, BarColumn, TextColumn
import threading

class BatchProcessor:
    def __init__(self, pipeline, max_workers: int = 4):
        self.pipeline = pipeline
        # CPU 密集型（OCR）不宜過多 thread；
        # GPU 模式可適度提高到 2~4
        self.max_workers = max_workers
        self._lock = threading.Lock()
        self.results = []
    
    def run(self, file_paths: list[Path]) -> list[ProcessingResult]:
        """批量執行，帶即時進度顯示"""
        self.results = []
        
        with Progress(
            SpinnerColumn(),
            "[progress.description]{task.description}",
            BarColumn(),
            "[progress.percentage]{task.percentage:>3.0f}%",
            "•",
            TextColumn("{task.completed}/{task.total} 份"),
        ) as progress:
            task = progress.add_task("處理文件中...", total=len(file_paths))
            
            with ThreadPoolExecutor(max_workers=self.max_workers) as executor:
                futures = {
                    executor.submit(process_with_retry, fp, self.pipeline): fp
                    for fp in file_paths
                }
                
                for future in as_completed(futures):
                    result = future.result()
                    with self._lock:
                        self.results.append(result)
                    progress.advance(task)
                    
                    status_icon = "✅" if result.status == Status.SUCCESS else "❌"
                    progress.log(
                        f"{status_icon} {Path(result.file_path).name} "
                        f"（{result.duration_sec}s）"
                    )
        
        return self.results
    
    def generate_batch_report(self) -> str:
        """批量處理摘要報告"""
        total = len(self.results)
        success = sum(1 for r in self.results if r.status == Status.SUCCESS)
        failed = sum(1 for r in self.results if r.status == Status.FAILED)
        
        report = f"""# 批量處理報告

## 摘要
- 總計：{total} 份
- 成功：{success} 份（{success/total*100:.1f}%）
- 失敗：{failed} 份
- 平均耗時：{sum(r.duration_sec for r in self.results)/total:.1f} 秒/份

## 失敗清單
"""
        for r in self.results:
            if r.status == Status.FAILED:
                report += f"- `{r.file_path}`\n  錯誤：{r.error[:200]}\n\n"
        
        return report
```
